// Model Wars + Hedge Your Bets — Google Apps Script // Handles two modes via GET parameters: // (no mode) — standard Model Wars: records chosen_id/chosen_name // mode=hedge — Hedge Your Bets: records bet_on_A and bet_on_B instead function doGet(e) { var params = e ? e.parameter : {}; if (params.student_id) { // Vote or bet submission var ss = SpreadsheetApp.getActiveSpreadsheet(); var mode = params.mode || 'vote'; if (mode === 'hedge') { recordHedge(ss, params); } else { recordVote(ss, params); } return ContentService.createTextOutput('ok'); } else { // Return totals JSON for leaderboard return getTotals(); } } // ── Standard Model Wars round ───────────────────────────────────────────────── function recordVote(ss, p) { var rounds = getOrCreate(ss, 'Rounds', [ 'timestamp','student_id','round', 'modelA_id','modelA_name','modelB_id','modelB_name', 'chosen_id','chosen_name' ]); rounds.appendRow([ new Date().toISOString(), p.student_id || '', parseInt(p.round) || 0, parseInt(p.modelA_id) || 0, p.modelA_name || '', parseInt(p.modelB_id) || 0, p.modelB_name || '', parseInt(p.chosen_id) || 0, p.chosen_name || '' ]); // Update Totals var totals = getOrCreate(ss, 'Totals', ['model_id','model_name','wins']); var chosenId = parseInt(p.chosen_id) || 0; var lastRow = totals.getLastRow(); var found = false; if (lastRow > 1) { var vals = totals.getRange(2,1,lastRow-1,3).getValues(); for (var i = 0; i < vals.length; i++) { if (parseInt(vals[i][0]) === chosenId) { totals.getRange(i+2,3).setValue((vals[i][2]||0)+1); found = true; break; } } } if (!found) totals.appendRow([chosenId, p.chosen_name||'', 1]); Logger.log('Vote: round ' + p.round + ' student ' + p.student_id + ' chose ' + chosenId); } // ── Hedge Your Bets round ───────────────────────────────────────────────────── function recordHedge(ss, p) { var hedgeSheet = getOrCreate(ss, 'Hedge_Rounds', [ 'timestamp','student_id','round', 'modelA_id','modelA_name','modelB_id','modelB_name', 'bet_on_A','bet_on_B' ]); hedgeSheet.appendRow([ new Date().toISOString(), p.student_id || '', parseInt(p.round) || 0, parseInt(p.modelA_id) || 0, p.modelA_name || '', parseInt(p.modelB_id) || 0, p.modelB_name || '', parseFloat(p.bet_on_A) || 0, parseFloat(p.bet_on_B) || 0 ]); // Update Hedge_Totals: cumulative dollars bet on each model var totals = getOrCreate(ss, 'Hedge_Totals', ['model_id','model_name','total_dollars']); updateHedgeTotals(totals, parseInt(p.modelA_id)||0, p.modelA_name||'', parseFloat(p.bet_on_A)||0); updateHedgeTotals(totals, parseInt(p.modelB_id)||0, p.modelB_name||'', parseFloat(p.bet_on_B)||0); Logger.log('Hedge: round ' + p.round + ' student ' + p.student_id + ' A=' + p.bet_on_A + ' B=' + p.bet_on_B); } function updateHedgeTotals(sheet, modelId, modelName, amount) { var lastRow = sheet.getLastRow(); if (lastRow > 1) { var vals = sheet.getRange(2,1,lastRow-1,3).getValues(); for (var i = 0; i < vals.length; i++) { if (parseInt(vals[i][0]) === modelId) { sheet.getRange(i+2,3).setValue((vals[i][2]||0) + amount); return; } } } sheet.appendRow([modelId, modelName, amount]); } // ── Return Totals JSON ──────────────────────────────────────────────────────── function getTotals() { var totals = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Totals'); var result = {}; if (totals && totals.getLastRow() > 1) { var data = totals.getRange(2,1,totals.getLastRow()-1,3).getValues(); for (var i = 0; i < data.length; i++) { if (data[i][0] !== '') result[String(data[i][0])] = data[i][2]||0; } } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ── Helper: get or create a sheet with headers ──────────────────────────────── function getOrCreate(ss, name, headers) { var sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(headers); sheet.getRange(1,1,1,headers.length).setFontWeight('bold'); sheet.setFrozenRows(1); } return sheet; } // ── Editor tests ────────────────────────────────────────────────────────────── function testVote() { var fake = { parameter: { student_id:'VOTETEST', round:'1', modelA_id:'1', modelA_name:'Model 1', modelB_id:'2', modelB_name:'Model 2', chosen_id:'1', chosen_name:'Model 1' }}; doGet(fake); Logger.log('testVote complete - check Rounds sheet'); } function testHedge() { var fake = { parameter: { mode:'hedge', student_id:'HEDGETEST', round:'1', modelA_id:'3', modelA_name:'Model 3', modelB_id:'5', modelB_name:'Model 5', bet_on_A:'0.70', bet_on_B:'0.30' }}; doGet(fake); Logger.log('testHedge complete - check Hedge_Rounds sheet'); }
